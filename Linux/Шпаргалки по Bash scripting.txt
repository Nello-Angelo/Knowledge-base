#!/bin/bash (начало скрипта)  скрипт запускается из папки, где был создан так    ./имя скрипта.sh
--------------------------------------------------------------------------
hello="hello world" (так создаеться текстовая переменная, которая содержит текст с пробелами и без. Чтобы Bash не обрабатывал содержимое, а воспринимал как текст, то нужно ставить одинарные ковычки или ставить знак доллар)
hello=helloworld, 100... (так создаеться переменная, которая не содержит пробелов, а просто значение)
--------------------------------------------------------------------------
if [ условие ]
then
    действия, которые дожны выполниться(команды и т.д.)
else
    альтернативное действие, которое должно выполниться вместо первого действия
fi (пишется в конце конструкции)
Таких конструкций может быть много в скрипте, как отдельных. так и вложенных друг в друга
Так же можно использовать после "if [ условие ] then действе " дополнитeльное условие elif [ условие ] then действе else альтернативное действие fi.
--------------------------------------------------------------------------
while [ условие ]
do 
   действие, которое будет выполняться, пока условие актуально
done (пишется в конце цикла)
--------------------------------------------------------------------------
nano /etc/systemd/system/название скрипта.service(в этот файл положим конфиг службы, это минимальная и простая конфигурация, этот файл будет демон-службой)(по этому пути самописные демоны)
подробнее по параметрам для конфига демона в документации systemd

[Unit]                                                 
Description=Description for echodate script
After=network.target

[Service]
Type=simple
ExecStart=/tmp/echodate.sh (путь до bash скрипта который создали раннее)
TimeoutStartSec=0

[Install]
WantedBy=default.target

systemctl daemon-reload (перезапускаем общую конфигурацию демона, потом стартуем его и добавляем в автозагрузку)
